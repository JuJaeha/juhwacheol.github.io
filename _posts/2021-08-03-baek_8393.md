import java.util.Scanner

fun main() {
    val sc: Scanner = Scanner(System.`in`)

    //1부터 n까지 +를 한다.

    var input_num = sc.nextInt()
    var result: Int = 0

    for (i in 1..input_num) {
        result = result + i
    }

    println(result)
}
