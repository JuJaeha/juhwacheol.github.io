//play.kotlinlang.org

import java.util.Scanner

fun main() {

    val sc: Scanner = Scanner(System.`in`)

    var num_a = sc.nextInt()
    var num_b = sc.nextInt()

    if (num_a > num_b) {
        println(">")
    } else if (num_a < num_b) {
        println("<")
    } else if (num_a == num_b) {
        println("==")
    }

}
