import java.util.Scanner

fun main() {
    val sc: Scanner = Scanner(System.`in`)

    var num1 : Int = sc.nextInt()
    var result: Int? = null

    if (num1 in 1..9) {
        for (b_num in 1..9) {
            result = num1 * b_num
            println("${num1} * ${b_num} = ${result!!}")
        }
    }
}
